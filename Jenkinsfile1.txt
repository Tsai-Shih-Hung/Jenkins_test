pipeline{
    agent any

    stages{
        stage('Hello'){
            steps{
                echo 'hello'
            }
        }
		stage('test version'){
            steps{
                sh '''
			    ansible --version
                ls -al
                pwd
                cat /etc/ansible/hosts
    		    '''
            }
        }
        stage('test ping'){
            steps{
                ansiblePlaybook become: true, 
                credentialsId: 'ansible', 
                disableHostKeyChecking: true, 
                installation: 'ansible', 
                inventory: '/etc/ansible/hosts', 
                playbook: 'test.yaml'
            }
        }
        stage('new vm'){
            steps{
                ansiblePlaybook become: true, 
                credentialsId: 'ansible', 
                disableHostKeyChecking: true, 
                extraVars: [
                    ansible_user:'root',
                    ansible_password:'123456'
                ], 
                installation: 'ansible', 
                inventory: '/etc/ansible/hosts', 
                playbook: 'addnewuser.yaml'
            }
        }
        stage('add inventory'){
            steps{
                sh'''
                echo '192.168.81.135' >> /etc/ansible/hosts
                cat /etc/ansible/hosts
                '''
            }
        }
        stage('check result'){
            steps{
               ansiblePlaybook become: true, 
               credentialsId: 'ansible', 
               disableHostKeyChecking: true, 
               installation: 'ansible', 
               inventory: '/etc/ansible/hosts', 
               playbook: 'ping.yaml'
            }
        }
    }
}
